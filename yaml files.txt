# Pod.yaml

apiVersion: v1
kind: Pod
metadata:
  name: banti-pod
  namespace: banti
  labels:
    app: nginx-app
spec:
  containers:
    - name: nginx-app
      image: nginx:latest
      ports:
        - containerPort: 80

# development.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: deployment-app
  namespace: banti
  labels:
    app: retail-app
spec:
  replicas: 3
  selector:
    matchLabels:
      app: retail-app
  template:
    metadata:
      labels:
        app: retail-app
    spec:
      containers:
        - name: my-container
          image: banti24/organic-ghee:latest
          ports:
            - containerPort: 80
